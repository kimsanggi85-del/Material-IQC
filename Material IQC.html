🔗 Supabase 설정
▶ Supabase SQL (테이블 생성) 
-- common_code 테이블 (설정값 저장)
CREATE TABLE IF NOT EXISTS common_code (
  id BIGSERIAL PRIMARY KEY,
  code_group VARCHAR(50) NOT NULL,
  code_key VARCHAR(200) NOT NULL,
  code_value TEXT,
  sort_order INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(code_group, code_key)
);

CREATE INDEX IF NOT EXISTS idx_common_code_group ON common_code(code_group);

-- RLS 활성화 (익명 읽기/쓰기 허용 - 내부망용)
ALTER TABLE common_code ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_all_common_code" ON common_code FOR ALL USING (true) WITH CHECK (true);

-- code_group 사용 예시:
--  mail_gas      : GAS 웹앱 URL
--  mat_mail      : 메일 수신자 목록
--  ntfy_topics   : ntfy 방 이름 목록

-- ========================================
-- 수입검사일보 테이블
-- ========================================
CREATE TABLE IF NOT EXISTS material_log (
  id TEXT PRIMARY KEY,
  date DATE,
  material VARCHAR(50),
  appear VARCHAR(10),
  thick_spec NUMERIC,
  width_spec NUMERIC,
  thick NUMERIC,
  width NUMERIC,
  in_kg NUMERIC,
  count INTEGER,
  bad_kg NUMERIC,
  heat VARCHAR(50),
  nc TEXT,
  label VARCHAR(20),
  judge VARCHAR(10),
  charge VARCHAR(50),
  boss VARCHAR(50),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_material_log_date ON material_log(date DESC);
ALTER TABLE material_log ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_all_material_log" ON material_log FOR ALL USING (true) WITH CHECK (true);

-- ========================================
-- 검사성적서 테이블 (공급자/수요자 측정값은 JSONB로)
-- ========================================
CREATE TABLE IF NOT EXISTS material_cert (
  id TEXT PRIMARY KEY,
  commodity TEXT,
  spec TEXT,
  surface TEXT,
  size TEXT,
  supplier TEXT,
  customer TEXT,
  date DATE,
  weight NUMERIC,
  cut TEXT,
  lot TEXT,
  heat TEXT,
  supplier_judge VARCHAR(10),
  supplier_sign TEXT,
  recv_date DATE,
  recv_charge TEXT,
  judge VARCHAR(10),
  customer_sign TEXT,
  label VARCHAR(20),
  remark TEXT,
  scan_id TEXT,
  measurements JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_material_cert_date ON material_cert(date DESC);
CREATE INDEX IF NOT EXISTS idx_material_cert_heat ON material_cert(heat);
ALTER TABLE material_cert ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_all_material_cert" ON material_cert FOR ALL USING (true) WITH CHECK (true);

-- ========================================
-- 자동 updated_at 갱신 트리거 (선택)
-- ========================================
CREATE OR REPLACE FUNCTION set_updated_at() RETURNS TRIGGER AS $$
BEGIN NEW.updated_at = NOW(); RETURN NEW; END;
$$ LANGUAGE plpgsql;
DROP TRIGGER IF EXISTS trg_log_updated ON material_log;
CREATE TRIGGER trg_log_updated BEFORE UPDATE ON material_log FOR EACH ROW EXECUTE FUNCTION set_updated_at();
DROP TRIGGER IF EXISTS trg_cert_updated ON material_cert;
CREATE TRIGGER trg_cert_updated BEFORE UPDATE ON material_cert FOR EACH ROW EXECUTE FUNCTION set_updated_at();
